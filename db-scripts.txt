CREATE TABLE `PORTFOLIO`
 ( `PORTFOLIO_ID` int(10) NOT NULL AUTO_INCREMENT, 
 `NAME` VARCHAR(255), 
 `CREATED_DATE` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
 PRIMARY KEY (`PORTFOLIO_ID`)
 );
 
 CREATE TABLE `PORTFOLIO_ITEM`
  ( 
  `PORTFOLIO_ITEM_ID` int(10) NOT NULL AUTO_INCREMENT, 
  `PORTFOLIO_ID` int(10) NOT NULL, 
  `SYMBOL` VARCHAR(10) NOT NULL, 
  `NOTE` VARCHAR(4000), 
  `CREATED_DATE` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
   
  PRIMARY KEY (`PORTFOLIO_ITEM_ID`),
  
  CONSTRAINT fk_portfolio FOREIGN KEY (`PORTFOLIO_ID`) REFERENCES PORTFOLIO (`PORTFOLIO_ID`),
  CONSTRAINT fk_mktlisting FOREIGN KEY (`SYMBOL`) REFERENCES MARKETLISTINGS (`SYMBOL`)
 );
 
 CREATE TABLE `marketlistings` (
   `Symbol` VARCHAR(10),
   `Name` varchar(1000),
   `LastSale` double DEFAULT NULL,
   `MarketCap` double DEFAULT NULL,
   `IPOyear` int(11) DEFAULT NULL,
   `Sector` varchar(1000),
   `industry` varchar(1000),
   `Summary Quote` varchar(1000)
   
   PRIMARY KEY (`Symbol`),
 );
